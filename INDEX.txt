// generate all scheduled session slots for the year
function generateSlots() {
  const slots = [];
  const start = parseDate(S.yearStart);
  const end   = parseDate(S.yearEnd);
  const cur   = new Date(start);
  while(cur <= end) {
    const iso  = isoDate(cur);
    const wday = cur.getDay(); // 0=sun,1=mon...
    if(wday>=1 && wday<=5 && !isFridag(iso)) {
      S.schedule.forEach(row => {
        if(row.dag === wday) {
          row.okter.forEach(okt => {
            slots.push({date:iso, fag:row.code, okt});
          });
        }
      });
    }
    cur.setDate(cur.getDate()+1);
  }
  return slots;
}

function renderPlanner() {
  buildFilters('planner-filters', S.plannerFilter, 'setPlannerFilter');
  const t = todayStr();
  const slots = generateSlots()
    .filter(s => s.date >= t)
    .filter(s => S.plannerFilter==='alle' || s.fag===S.plannerFilter);

  // also add any extra sessions
  const extras = Object.keys(S.sessions)
    .filter(k => S.sessions[k].extra)
    .map(k => { const [date,fag,okt]=k.split('|'); return {date,fag,okt:+okt}; })
    .filter(s => s.date >= t)
    .filter(s => S.plannerFilter==='alle' || s.fag===S.plannerFilter);

  const all = [...slots, ...extras].sort((a,b)=>a.date.localeCompare(b.date)||a.okt-b.okt);

  // deduplicate
  const seen = new Set();
  const deduped = all.filter(s => { const k=`${s.date}|${s.fag}|${s.okt}`; if(seen.has(k)) return false; seen.add(k); return true; });

  const container = document.getElementById('planner-content');
  if(!deduped.length) { container.innerHTML=`<div class="empty-state">Ingen kommende økter. Legg til fag i Tilgang &amp; timeplan.</div>`; return; }

  const byDate = {};
  deduped.forEach(s => { if(!byDate[s.date]) byDate[s.date]=[]; byDate[s.date].push(s); });

  let html = '';
  Object.keys(byDate).sort().forEach(date => {
    const evs = S.events.filter(e=>e.date===date);
    html += `<div class="day-group"><div class="day-header"><div class="day-label">${fmtDate(date)}</div>`;
    evs.forEach(e => {
      const cls = e.type==='fridag'?'fridag':e.type==='nb'?'nb':'';
      html += `<div class="day-merkedag ${cls}">${e.title}</div>`;
    });
    html += `<div class="day-line"></div></div>`;
    byDate[date].forEach(s => {
      const k = `${s.date}|${s.fag}|${s.okt}`;
      const sess = getSession(s.date, s.fag, s.okt);
      const hasContent = sess.content.trim().length > 0;
      const preview = hasContent ? sess.content.substring(0,60)+(sess.content.length>60?'…':'') : 'Ikke planlagt ennå';
      const hasCollab = sess.collabNotes && sess.collabNotes.length > 0;
      const isExtra = sess.extra;
      html += `
        <div class="session" id="p-${k.replace(/\|/g,'-')}">
          <div class="session-header" onclick="toggleSess('p-${k.replace(/\|/g,'-')}')">
            <div class="okt-time">${oktLabel(s.okt)}</div>
            <div class="subject-tag ${tagCls(s.fag)}">${s.fag}${isExtra?' ✦':''}</div>
            <div class="session-preview${hasContent?' has-content':''}">${preview}</div>
            ${hasCollab?`<span style="font-size:10px;color:var(--tip);margin-right:4px;">kollega</span>`:''}
            <svg class="chevron" viewBox="0 0 12 12" fill="none"><path d="M4 2l4 4-4 4" stroke="currentColor" stroke-width="1.2" stroke-linecap="round" stroke-linejoin="round"/></svg>
          </div>
          <div class="session-body">
            <div class="session-divider"></div>
            <div class="field-label">Innhold</div>
            <textarea class="field-input" id="c-${k.replace(/\|/g,'-')}" placeholder="Hva skal skje i denne økten?">${sess.content}</textarea>
            <div class="private-block">
              <div class="private-label-tag">Privat notat</div>
              <textarea class="field-input" id="n-${k.replace(/\|/g,'-')}" placeholder="Kun synlig for deg..." style="min-height:44px;">${sess.note}</textarea>
            </div>
            ${hasCollab ? `
            <div class="collab-block">
              <div class="collab-label-tag">Kollega-notat — ${s.fag}</div>
              <div class="collab-thread">
                ${sess.collabNotes.map(m=>`
                  <div class="collab-msg">
                    <div class="collab-msg-author${m.author==='Meg'?' mine':''}">${m.author}</div>
                    <div class="collab-msg-text">${m.text}</div>
                  </div>`).join('')}
              </div>
              <textarea class="field-input" id="cn-${k.replace(/\|/g,'-')}" placeholder="Svar til kollega..." style="min-height:40px;"></textarea>
              <button class="btn btn-sm" style="margin-top:6px;" onclick="sendCollabNote('${k}')">Send</button>
            </div>` : ''}
            <div class="session-actions">
              <button class="btn btn-primary btn-sm" onclick="saveEdit('${k}')">Lagre</button>
              ${isExtra?`<button class="btn btn-sm" onclick="deleteExtra('${k}')">Slett</button>`:''}
            </div>
          </div>
        </div>`;
    });
    html += `</div>`;
  });
  container.innerHTML = html;
}

async function deleteEvent(id) {
  if(!id) return;
  const deleted = S.events.find(e => String(e.id) === String(id));
  S.events = S.events.filter(e => String(e.id) !== String(id));
  renderCalendar();
  if(deleted) {
    pushUndo('Slett ' + deleted.title, async () => {
      S.events.push(deleted);
      renderCalendar();
      if(USE_BACKEND) await apiCall('saveEvent', deleted);
    });
  }
  if(USE_BACKEND) {
    showSyncStatus('Sletter...');
    await apiCall('deleteEvent', {id});
    showSyncStatus('Slettet');
  }
}

const MONTH_NMS = ['januar','februar','mars','april','mai','juni','juli','august','september','oktober','november','desember'];
const DAY_NMS_SHORT = ['ma','ti','on','to','fr','lø','sø'];
let calYear, calMonth;

function calInit() {
  const n = new Date();
  calYear = n.getFullYear();
  calMonth = n.getMonth();
  drawCals();
}

function calPrev() {
  calMonth--;
  if(calMonth < 0) { calMonth = 11; calYear--; }
  drawCals();
}

function calNext() {
  calMonth++;
  if(calMonth > 11) { calMonth = 0; calYear++; }
  drawCals();
}

function isHoliday(iso) {
  return S.events.some(e => e.date === iso && e.type === 'fridag');
}
function hasEvent(iso) {
  return S.events.some(e => e.date === iso && e.type !== 'fridag');
}
function hasSession(iso) {
  try { return generateSlots().some(s => s.date === iso); } catch(e) { return false; }
}
function isoFromYMD(y,m,d) {
  return y+'-'+String(m+1).padStart(2,'0')+'-'+String(d).padStart(2,'0');
}

function drawCals() {
  drawMiniCal();
  drawFullCal();
}

function drawMiniCal() {
  // Draw into sidebar version (if exists) and right-column version
  _drawMiniCalInto('mini-cal-title', 'mini-cal-grid');
  _drawMiniCalInto('mini-cal-title-2', 'mini-cal-grid-2');
  drawUpcomingEvents();
}

function _drawMiniCalInto(titleId, gridId) {
  const titleEl = document.getElementById(titleId);
  const gridEl  = document.getElementById(gridId);
  if(!titleEl || !gridEl) return;

  titleEl.textContent = MONTH_NMS[calMonth].substring(0,3) + ' ' + calYear;

  const today = new Date();
  const todayIso = isoFromYMD(today.getFullYear(), today.getMonth(), today.getDate());

  // day name headers
  let html = DAY_NMS_SHORT.map(d => `<div class="cal-day-name">${d}</div>`).join('');

  const first = new Date(calYear, calMonth, 1);
  let offset = first.getDay() - 1;
  if(offset < 0) offset = 6;

  const daysInMonth = new Date(calYear, calMonth+1, 0).getDate();
  const prevDays = new Date(calYear, calMonth, 0).getDate();

  for(let i = offset-1; i >= 0; i--) {
    html += `<div class="cal-day other-month">${prevDays-i}</div>`;
  }

  for(let d = 1; d <= daysInMonth; d++) {
    const iso = isoFromYMD(calYear, calMonth, d);
    const wday = new Date(calYear, calMonth, d).getDay();
    const weekend = wday===0||wday===6;
    let cls = 'cal-day';
    if(iso === todayIso) cls += ' is-today';
    if(isHoliday(iso)) cls += ' is-holiday';
    else if(hasEvent(iso)) cls += ' has-dot has-event';
    else if(hasSession(iso)) cls += ' has-dot';
    else if(weekend) cls += ' is-holiday';
    html += `<div class="${cls}">${d}</div>`;
  }

  const total = offset + daysInMonth;
  const rem = total%7===0?0:7-(total%7);
  for(let d=1;d<=rem;d++) html += `<div class="cal-day other-month">${d}</div>`;

  gridEl.innerHTML = html;
}

  const t = todayStr();
  const upcoming = S.events

    .slice(0, 5);
  if(!upcoming.length) { container.innerHTML = ''; return; }
  const typeColor = {fridag:'#5a5650',forestilling:'#EF9F27',eksamen:'#E24B4A',aktivitet:'#5DCAA5',nb:'#AFA9EC'};
  let html = '<div style="font-size:9px;color:var(--text3);letter-spacing:0.1em;text-transform:uppercase;margin-bottom:8px;">Kommende</div>';
  upcoming.forEach(e => {
    const col = typeColor[e.type] || '#888';
    html += `<div style="display:flex;gap:8px;align-items:flex-start;margin-bottom:8px;">
      <div style="width:5px;height:5px;border-radius:50%;background:${col};margin-top:4px;flex-shrink:0;"></div>
      <div><div style="font-size:11px;color:var(--text2);">${e.title}</div><div style="font-size:10px;color:var(--text3);">${fmtDate(e.date)}</div></div>
    </div>`;
  });
  container.innerHTML = html;
}

function drawFullCal() {
  const titleEl = document.getElementById('full-cal-title');
  const gridEl  = document.getElementById('full-cal-grid');
  if(!titleEl || !gridEl) return;

  titleEl.textContent = MONTH_NMS[calMonth] + ' ' + calYear;

  const today = new Date();
  const todayIso = isoFromYMD(today.getFullYear(), today.getMonth(), today.getDate());
  let slots = [];
  try { slots = generateSlots(); } catch(e) {}

  const DAY_LONG = ['Man','Tir','Ons','Tor','Fre','Lør','Søn'];
  let html = DAY_LONG.map(d=>`<div class="full-cal-name">${d}</div>`).join('');

  const first = new Date(calYear, calMonth, 1);
  let offset = first.getDay()-1;
  if(offset < 0) offset = 6;

  const daysInMonth = new Date(calYear, calMonth+1, 0).getDate();

  for(let i=0;i<offset;i++) html += `<div class="full-cal-cell other-month"></div>`;

  const fagColors = {TOB:'#AFA9EC',TIP:'#5DCAA5',TEP:'#EF9F27',KON:'#9ac8ef'};

  for(let d=1;d<=daysInMonth;d++) {
    const iso = isoFromYMD(calYear, calMonth, d);
    const isToday = iso===todayIso;
    const wday = new Date(calYear,calMonth,d).getDay();
    const weekend = wday===0||wday===6;
    const holiday = isHoliday(iso);
    const evs = S.events.filter(e=>e.date===iso&&e.type!=='fridag');
    const daySessions = slots.filter(s=>s.date===iso);
    const subjects = [...new Set(daySessions.map(s=>s.fag))];

    let cls='full-cal-cell';
    if(isToday) cls+=' is-today';
    if(holiday) cls+=' is-holiday';
    else if(weekend && evs.length===0) cls+=' is-holiday';

    let inner = `<div class="full-cal-date">${d}</div>`;
    evs.forEach(e=>{inner+=`<div class="full-cal-chip">${e.title}</div>`;});
    subjects.forEach(fag=>{
      const key=fag.replace(/\d/g,'').toUpperCase();
      const col=fagColors[key]||'#888';
      inner+=`<div class="full-cal-subj" style="color:${col}">${fag}</div>`;
    });
    html+=`<div class="${cls}">${inner}</div>`;
  }

  const total=offset+daysInMonth;
  const rem=total%7===0?0:7-(total%7);
  for(let i=0;i<rem;i++) html+=`<div class="full-cal-cell other-month"></div>`;

  gridEl.innerHTML=html;
}

  }

  // Falls back to direct call if not in Apps Script (local testing)
  if(typeof google !== 'undefined' && google.script) {
    google.script.run
      .withSuccessHandler(function() { calInit(); })
      .withFailureHandler(function() { calInit(); })
      .getAppVersion();
  } else {
    calInit();
  }
  checkSharedAccess().then(isShared => { if(!isShared) loadFromBackend(); });
};

// Set this to your deployed Apps Script URL after publishing
const CALENDAR_URL = ''; // ← fyll inn URL til Kalender.html når den er deployet
const API_URL = 'DIN_APPS_SCRIPT_URL_HER';
const USE_BACKEND = API_URL !== 'DIN_APPS_SCRIPT_URL_HER';

async function apiCall(action, data={}) {
  if(!USE_BACKEND) return null; // use local state only
  try {
    const res = await fetch(API_URL, {
      method: 'POST',
      body: JSON.stringify({action, data}),
    });
    const json = await res.json();
    if(!json.ok) throw new Error(json.error);
    return json.data;
  } catch(err) {
    console.warn('API feil:', err.message);
    return null;
  }
}

async function loadFromBackend() {
  if(!USE_BACKEND) return;
  showSyncStatus('Laster...');
  const all = await apiCall('getAll');
  if(!all) { showSyncStatus('Frakoblet'); return; }

  // settings
  if(all.settings) {
    if(all.settings.yearStart) S.yearStart = all.settings.yearStart;
    if(all.settings.yearEnd)   S.yearEnd   = all.settings.yearEnd;
  }
  // schedule
  if(all.schedule && all.schedule.length) S.schedule = all.schedule;
  // events
  if(all.events)  S.events  = all.events;
  // sessions
  if(all.sessions) S.sessions = all.sessions;
  // access
  if(all.access)  S.access  = all.access;

  showSyncStatus('Synkronisert');
  renderPlanner();
}

function showSyncStatus(msg) {
  let el = document.getElementById('sync-status');
  if(!el) return;
  el.textContent = msg;
  el.style.opacity = '1';
  if(msg === 'Synkronisert') {
    setTimeout(() => { el.style.opacity = '0'; }, 2000);
  }
}

const _origSaveEdit = saveEdit;
saveEdit = async function(k) {
  // snapshot before save for undo
  const before = {...getSession(...k.split('|'))};
  const beforeContent = before.content;
  const beforeNote = before.note;
  _origSaveEdit(k);
  pushUndo('Lagre økt', async () => {
    const sess = getSession(...k.split('|'));
    sess.content = beforeContent;
    sess.note = beforeNote;
    renderPlanner();
    if(USE_BACKEND) await apiCall('saveSession', {key:k, ...sess});
  });
  if(USE_BACKEND) {
    const sess = getSession(...k.split('|'));
    showSyncStatus('Lagrer...');
    await apiCall('saveSession', {key:k, ...sess});
    showSyncStatus('Lagret');
  }
};

// ── URL ROUTING ───────────────────────────────────────────────────────────────
// When a student/guest visits a shared link, the URL contains a token
// e.g. ?token=abc123 — we check this against the access list and show
// the appropriate filtered view instead of the full planner.

function getUrlParam(name) {
  const url = window.location.href;
  const match = url.match(new RegExp('[?&]' + name + '=([^&]+)'));
  return match ? decodeURIComponent(match[1]) : null;
}

async function checkSharedAccess() {
  const token = getUrlParam('token');
  if(!token) return false; // no token = teacher view, load normally

  // load all data first
  if(USE_BACKEND) {
    const all = await apiCall('getAll');
    if(all) {
      if(all.settings) {
        if(all.settings.yearStart) S.yearStart = all.settings.yearStart;
        if(all.settings.yearEnd)   S.yearEnd   = all.settings.yearEnd;
      }
      if(all.schedule && all.schedule.length) S.schedule = all.schedule;
      if(all.events)  S.events  = all.events;
      if(all.sessions) S.sessions = all.sessions;
      if(all.access)  S.access  = all.access;
    }
  }

  // find matching access entry by token
  const entry = S.access.find(a => a.active && generateToken(a.id) === token);
  if(!entry) {
    document.body.innerHTML = '<div style="padding:40px;font-family:monospace;color:#888;text-align:center;background:#0e0d0c;min-height:100vh;display:flex;align-items:center;justify-content:center;"><div><div style="font-size:14px;margin-bottom:8px;">Lenken er ikke aktiv.</div><div style="font-size:12px;color:#555;">Ta kontakt med læreren din.</div></div></div>';
    return true;
  }

  // parse subject list from entry
  const subjects = entry.fag.split(',').map(s => s.trim()).filter(Boolean);

  // hide sidebar, show student-style view
  document.querySelector('.sidebar').style.display = 'none';
  document.querySelector('.main').style.marginLeft = '0';

  // inject student name/school header
  const nameEl = document.getElementById('student-name-display');
  if(nameEl) nameEl.textContent = entry.name;
  const subEl = document.getElementById('student-subject-display');
  if(subEl) subEl.textContent = subjects.join(', ');

  // override STUDENT_FAGS with this entry's subjects
  window.SHARED_SUBJECTS = subjects;
  window.SHARED_ROLE = entry.role;
  window.SHARED_ENTRY = entry;

  // show shared view
  showSharedView(entry, subjects);
  return true;
}

function generateToken(id) {
  // simple deterministic token from id — in production use a proper hash
  // for now we use the id itself encoded
  return btoa(id).replace(/=/g,'').replace(/\+/g,'-').replace(/\//g,'_');
}

function getShareUrl(entry) {
  const base = window.location.href.split('?')[0];
  const token = generateToken(entry.id);
  return base + '?token=' + token;
}

function showSharedView(entry, subjects) {

  if(shared) shared.classList.add('active');

  // build nav tabs for colleague (sessions + calendar), just filters for others
  const topbar = shared.querySelector('.topbar');
  if(isKollega) {
    topbar.innerHTML = `
      <div>
        <div class="view-title" id="student-name-display">${entry.name}</div>
        <div style="font-size:11px;color:var(--text3);margin-top:2px;">Faglærer: Ren Powell &nbsp;·&nbsp; Vågen vgs</div>
      </div>
      <div style="display:flex;gap:8px;align-items:center;">
        <button class="btn${window._sharedTab!=='kalender'?' btn-primary':''}" id="tab-okter" onclick="setSharedTab('okter')">Mine økter</button>
        <button class="btn${window._sharedTab==='kalender'?' btn-primary':''}" id="tab-kalender" onclick="setSharedTab('kalender')">Merkedager</button>
      </div>`;
    // add filter row below topbar if not already there
    let filterRow = document.getElementById('shared-filters-row');
    if(!filterRow) {
      filterRow = document.createElement('div');
      filterRow.id = 'shared-filters-row';
      filterRow.className = 'filter-row';
      filterRow.style.cssText = 'padding:10px 32px;border-bottom:0.5px solid var(--border);';
      shared.insertBefore(filterRow, shared.querySelector('.content, div[style*="max-width"]'));
    }
    filterRow.id = 'shared-filters';
  } else {
    topbar.innerHTML = `
      <div>
        <div class="view-title" id="student-name-display">${entry.name}</div>
        <div style="font-size:11px;color:var(--text3);margin-top:2px;">Faglærer: Ren Powell &nbsp;·&nbsp; Vågen vgs</div>
      </div>
      <div class="filter-row" id="shared-filters"></div>`;
  }

  // build subject filter pills
  const row = document.getElementById('shared-filters');
  if(row) {
    let html = `<button class="pill active" onclick="setSharedFilter('alle',this)">Alle fag</button>`;
    subjects.forEach(c => {
      const cls = c.replace(/\d/g,'').toLowerCase();
      html += `<button class="pill ${cls}" onclick="setSharedFilter('${c}',this)">${c}</button>`;
    });
    row.innerHTML = html;
  }

  window._sharedFilter = 'alle';
  window._sharedTab = 'okter';
  renderSharedSessions(subjects, 'alle');
}

function setSharedTab(tab) {
  window._sharedTab = tab;
  const btnOkter = document.getElementById('tab-okter');
  const btnKal   = document.getElementById('tab-kalender');
  if(btnOkter) { btnOkter.className = tab==='okter' ? 'btn btn-primary' : 'btn'; }
  if(btnKal)   { btnKal.className   = tab==='kalender' ? 'btn btn-primary' : 'btn'; }

  const filtersRow = document.getElementById('shared-filters');
  if(filtersRow) filtersRow.style.display = tab==='okter' ? 'flex' : 'none';

  const container = document.getElementById('shared-content');
  if(tab === 'kalender') {
    renderSharedCalendar(container);
  } else {
    renderSharedSessions(window.SHARED_SUBJECTS, window._sharedFilter || 'alle');
  }
}

function renderSharedCalendar(container) {
  if(!container) return;
  const sorted = [...S.events].sort((a,b) => a.date.localeCompare(b.date));
  if(!sorted.length) {
    container.innerHTML = `
      <div style="padding:16px 0 8px;">
        <div class="cal-legend" style="padding:0 0 16px;">
          <div class="legend-item"><div class="legend-dot" style="background:#5a5650"></div>Fridag</div>
          <div class="legend-item"><div class="legend-dot" style="background:#EF9F27"></div>Forestilling</div>
          <div class="legend-item"><div class="legend-dot" style="background:#E24B4A"></div>Eksamen</div>
          <div class="legend-item"><div class="legend-dot" style="background:#5DCAA5"></div>Aktivitet</div>
          <div class="legend-item"><div class="legend-dot" style="background:#AFA9EC"></div>NB</div>
        </div>
      </div>
      <div class="empty-state">Ingen merkedager ennå.</div>`;
    return;
  }
  const byMonth = {};
  sorted.forEach(e => { const m=fmtMonth(e.date); if(!byMonth[m]) byMonth[m]=[]; byMonth[m].push(e); });
  const typeColor = {fridag:'#5a5650',forestilling:'#EF9F27',eksamen:'#E24B4A',aktivitet:'#5DCAA5',nb:'#AFA9EC'};

  let html = `
    <div style="padding:16px 0 8px;">
      <div style="display:flex;justify-content:flex-end;margin-bottom:12px;">
        <button class="btn btn-primary btn-sm" onclick="openSharedEventModal()">+ Legg til dato</button>
      </div>
    </div>`;

  Object.keys(byMonth).forEach(month => {
    html += `<div class="day-group"><div class="cal-month-header">${month}<div class="day-line"></div></div>`;
    byMonth[month].forEach(e => {
      html += `
        <div class="cal-event">
          <div class="cal-date">${fmtDate(e.date)}</div>
          <div class="cal-dot" style="background:${typeColor[e.type]||'#888'};margin-top:5px;flex-shrink:0;width:7px;height:7px;border-radius:50%;"></div>
          <div class="cal-info"><div class="cal-title">${e.title}</div>${e.fag?`<div class="cal-sub">${e.fag}</div>`:''}</div>
          <div class="cal-badge badge-${e.type}">${e.type}</div>
          <button class="toggle-btn btn-sm" data-event-id="${e.id}" onclick="deleteSharedEvent(this.dataset.eventId)" style="margin-left:8px;">×</button>
        </div>`;
    });
    html += `</div>`;
  });
  container.innerHTML = html;
}

async function deleteSharedEvent(id) {
  if(!id) return;
  const deleted = S.events.find(e => String(e.id) === String(id));
  S.events = S.events.filter(e => String(e.id) !== String(id));
  renderSharedCalendar(document.getElementById('shared-content'));
  if(deleted) {
    pushUndo('Slett ' + deleted.title, async () => {
      S.events.push(deleted);
      renderSharedCalendar(document.getElementById('shared-content'));
      if(USE_BACKEND) await apiCall('saveEvent', deleted);
    });
  }
  if(USE_BACKEND) {
    showSyncStatus('Sletter...');
    await apiCall('deleteEvent', {id});
    showSyncStatus('Slettet');
  }
}

function openSharedEventModal() {
  const now = new Date();
  const y = now.getFullYear();
  const m = String(now.getMonth()+1).padStart(2,'0');
  const d = String(now.getDate()).padStart(2,'0');
  document.getElementById('ev-date').value = y+'-'+m+'-'+d;
  document.getElementById('ev-title').value = '';
  document.getElementById('ev-fag').value = '';
  document.getElementById('ev-type').value = 'fridag';
  // override save to re-render shared calendar
  window._saveEventCallback = () => renderSharedCalendar(document.getElementById('shared-content'));
  document.getElementById('modal-event').classList.add('open');
}

function setSharedFilter(f, el) {
  window._sharedFilter = f;
  document.querySelectorAll('#shared-filters .pill').forEach(p => p.classList.remove('active'));
  el.classList.add('active');
  renderSharedSessions(window.SHARED_SUBJECTS, f);
}

function renderSharedSessions(subjects, filter) {
  const t = todayStr();
  const isKollega = window.SHARED_ROLE === 'kollega';

  let slots = generateSlots()
    .filter(s => s.date >= t)
    .filter(s => subjects.includes(s.fag))
    .filter(s => filter === 'alle' || s.fag === filter)
    .sort((a,b) => a.date.localeCompare(b.date) || a.okt - b.okt);

  const container = document.getElementById('shared-content');
  if(!container) return;
  if(!slots.length) { container.innerHTML = `<div class="empty-state">Ingen kommende økter.</div>`; return; }

  const byDate = {};
  slots.forEach(s => { if(!byDate[s.date]) byDate[s.date]=[]; byDate[s.date].push(s); });

  let html = '';
  Object.keys(byDate).sort().forEach(date => {
    const evs = S.events.filter(e => e.date===date && (!e.fag || subjects.some(f=>e.fag.includes(f))));
    html += `<div class="day-group"><div class="day-header"><div class="day-label">${fmtDate(date)}</div>`;
    evs.forEach(e => html += `<div class="day-merkedag">${e.title}</div>`);
    html += `<div class="day-line"></div></div>`;
    byDate[date].forEach(s => {
      const sess = getSession(s.date, s.fag, s.okt);
      const hasContent = sess.content.trim().length > 0;
      const k = `sh-${s.date}-${s.fag}-${s.okt}`;
      const collabNotes = sess.collabNotes || [];
      html += `
        <div class="session" id="${k}">
          <div class="session-header" onclick="toggleSess('${k}')">
            <div class="okt-time">${oktLabel(s.okt)}</div>
            <div class="subject-tag ${tagCls(s.fag)}">${s.fag}</div>
            <div class="session-preview${hasContent?' has-content':''}">${hasContent ? sess.content.substring(0,60)+(sess.content.length>60?'…':'') : 'Ikke planlagt ennå'}</div>
            <svg class="chevron" viewBox="0 0 12 12" fill="none"><path d="M4 2l4 4-4 4" stroke="currentColor" stroke-width="1.2" stroke-linecap="round" stroke-linejoin="round"/></svg>
          </div>
          <div class="session-body">
            <div class="session-divider"></div>
            ${isKollega ? `
              <div class="field-label">Innhold</div>
              <textarea class="field-input" id="sc-${s.date}-${s.fag}-${s.okt}" placeholder="Hva skal skje?">${sess.content}</textarea>
              <div class="session-actions">
                <button class="btn btn-primary btn-sm" onclick="saveSharedEdit('${s.date}','${s.fag}',${s.okt})">Lagre</button>
              </div>
              ${collabNotes.length ? `
              <div class="collab-block" style="margin-top:10px;">
                <div class="collab-label-tag">Notat</div>
                ${collabNotes.map(m=>`<div class="collab-msg"><div class="collab-msg-author">${m.author}</div><div class="collab-msg-text">${m.text}</div></div>`).join('')}
              </div>` : ''}
              <div class="collab-block" style="margin-top:8px;">
                <div class="collab-label-tag">Skriv notat til lærer</div>
                <textarea class="field-input" id="cn-sh-${s.date}-${s.fag}-${s.okt}" placeholder="Notat..." style="min-height:40px;"></textarea>
                <button class="btn btn-sm" style="margin-top:6px;" onclick="sendSharedCollab('${s.date}','${s.fag}',${s.okt})">Send</button>
              </div>
            ` : `
              <div class="field-text${hasContent?'':' empty'}">${hasContent ? sess.content : 'Ingen beskrivelse lagt til ennå.'}</div>
              ${evs.length ? '' : ''}
            `}
          </div>
        </div>`;
    });
    html += `</div>`;
  });
  container.innerHTML = html;
}

async function saveSharedEdit(date, fag, okt) {
  const k = `${date}|${fag}|${okt}`;
  const sess = getSession(date, fag, okt);
  const el = document.getElementById(`sc-${date}-${fag}-${okt}`);
  if(el) sess.content = el.value;
  if(USE_BACKEND) {
    showSyncStatus('Lagrer...');
    await apiCall('saveSession', {key:k, ...sess});
    showSyncStatus('Lagret');
  }
  renderSharedSessions(window.SHARED_SUBJECTS, window._sharedFilter || 'alle');
}

async function sendSharedCollab(date, fag, okt) {
  const k = `${date}|${fag}|${okt}`;
  const sess = getSession(date, fag, okt);
  const el = document.getElementById(`cn-sh-${date}-${fag}-${okt}`);
  if(!el || !el.value.trim()) return;
  const author = window.SHARED_ENTRY ? window.SHARED_ENTRY.name : 'Kollega';
  sess.collabNotes.push({author, text: el.value.trim()});
  if(USE_BACKEND) await apiCall('saveSession', {key:k, ...sess});
  renderSharedSessions(window.SHARED_SUBJECTS, window._sharedFilter || 'alle');
}

  });
}
